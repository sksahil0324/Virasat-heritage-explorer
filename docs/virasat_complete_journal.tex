\subsection{Image Management System}
Admins can manage site images by deleting unwanted images and selecting a primary image for display.

\subsubsection{Backend Mutations}
\begin{lstlisting}[caption=Image Management Mutations]
// Remove media
export const remove = mutation({
  args: { id: v.id("media") },
  handler: async (ctx, args) => {
    const user = await getCurrentUser(ctx);
    if (!user || user.role !== "admin") {
      throw new Error("Unauthorized");
    }
    await ctx.db.delete(args.id);
  },
});

// Set media as primary
export const setPrimary = mutation({
  args: { 
    id: v.id("media"),
    siteId: v.id("heritageSites")
  },
  handler: async (ctx, args) => {
    const user = await getCurrentUser(ctx);
    if (!user || user.role !== "admin") {
      throw new Error("Unauthorized");
    }
    // Get all media for this site
    const siteMedia = await ctx.db
      .query("media")
      .withIndex("by_site", (q) => q.eq("siteId", args.siteId))
      .collect();
    // Set all to non-primary
    for (const media of siteMedia) {
      await ctx.db.patch(media._id, { isPrimary: false });
    }
    // Set the selected one as primary
    await ctx.db.patch(args.id, { isPrimary: true });
  },
});
\end{lstlisting}

\subsubsection{Frontend Image Management UI}
The admin dashboard displays all uploaded images in a grid with hover actions:
\begin{itemize}
    \item \textbf{Set as Primary:} Click the star icon to designate an image as the primary display image
    \item \textbf{Delete Image:} Click the trash icon to remove unwanted images
    \item \textbf{Visual Indicator:} Primary images are highlighted with a blue border and "Primary" badge
\end{itemize}

\subsubsection{Image Management Algorithm}
\begin{algorithm}[H]
\caption{Image Management Process}
\begin{algorithmic}[1]
\State Display all media for the site in a grid layout
\For{each media item}
    \If{media.isPrimary}
        \State Show "Primary" badge and blue border
    \EndIf
    \On{hover}
        \If{not isPrimary}
            \State Show star button to set as primary
        \EndIf
        \State Show trash button to delete
    \EndOn
\EndFor
\On{set as primary click}
    \State Fetch all media for the site
    \State Set all media isPrimary to false
    \State Set selected media isPrimary to true
    \State Show success toast notification
\EndOn
\On{delete click}
    \State Confirm deletion with user
    \State Delete media record from database
    \State Show success toast notification
\EndOn
\end{algorithmic}
\end{algorithm}
